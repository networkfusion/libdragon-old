766inline void rdpq_set_tile(rdpq_tile_t tile,
767 tex_format_t format,
768 int32_t tmem_addr,
769 uint16_t tmem_pitch,
770 const rdpq_tileparms_t *parms)
771{
772 static const rdpq_tileparms_t default_parms = {0};
773 if (!parms) parms = &default_parms;
774 else {
775 assertf(parms->s.shift >= -5 && parms->s.shift <= 10, "invalid s shift %d: must be in [-5..10]", parms->s.shift);
776 assertf(parms->t.shift >= -5 && parms->t.shift <= 10, "invalid t shift %d: must be in [-5..10]", parms->t.shift);
777 }
778 bool fixup = false;
779 bool reuse = false;
780 uint32_t cmd_id = RDPQ_CMD_SET_TILE;
781 if (tmem_addr & (RDPQ_AUTOTMEM | RDPQ_AUTOTMEM_REUSE(0))) {
782 cmd_id = RDPQ_CMD_AUTOTMEM_SET_TILE;
783 reuse = (tmem_addr & RDPQ_AUTOTMEM_REUSE(0)) != 0;
784 fixup = true;
785 tmem_addr &= ~(RDPQ_AUTOTMEM | RDPQ_AUTOTMEM_REUSE(0));
786 } else {
787 assertf((tmem_addr % 8) == 0, "invalid tmem_addr %ld: must be multiple of 8", tmem_addr);
788 tmem_addr /= 8;
789 }
790 assertf((tmem_pitch % 8) == 0, "invalid tmem_pitch %d: must be multiple of 8", tmem_pitch);
791 extern void __rdpq_write8_syncchange(uint32_t, uint32_t, uint32_t, uint32_t);
792 extern void __rdpq_fixup_write8_syncchange(uint32_t, uint32_t, uint32_t, uint32_t);
793 (fixup ? __rdpq_fixup_write8_syncchange : __rdpq_write8_syncchange)(cmd_id,
794 _carg(format, 0x1F, 19) | _carg(reuse, 0x1, 18) | _carg(tmem_pitch/8, 0x1FF, 9) | _carg(tmem_addr, 0x1FF, 0),
795 _carg(tile, 0x7, 24) | _carg(parms->palette, 0xF, 20) |
796 _carg(parms->t.clamp | (parms->t.mask == 0), 0x1, 19) | _carg(parms->t.mirror, 0x1, 18) | _carg(parms->t.mask, 0xF, 14) | _carg(parms->t.shift, 0xF, 10) |
797 _carg(parms->s.clamp | (parms->s.mask == 0), 0x1, 9) | _carg(parms->s.mirror, 0x1, 8) | _carg(parms->s.mask, 0xF, 4) | _carg(parms->s.shift, 0xF, 0),
798 AUTOSYNC_TILE(tile));
799}
